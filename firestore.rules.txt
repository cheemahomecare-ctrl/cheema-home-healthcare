rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // MVP rules: anyone can create a booking, but only people who know
    // this app's admin passcode (checked in the app, not here) can see/manage them.
    // For a production-grade app, add real authentication later.
    match /bookings/{bookingId} {
      allow create: if true;
      allow read, update, delete: if true;
    }
    match /nurses/{nurseId} {
      allow read, write: if true;
    }
  }
}

// NOTE: these rules are intentionally open so the app works without login.
// Anyone with your Firebase project URL could technically read/write data.
// This is fine for an early MVP but should be tightened (e.g. with Firebase
// Auth for the admin dashboard) before handling sensitive patient data at scale.
